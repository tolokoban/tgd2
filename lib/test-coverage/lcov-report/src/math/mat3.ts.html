/**
 * Column-first 4x4 matrix.
 *
 * `m23` means column 2 and row 3.
 *
 * This is not the same as what mathematicians are used to,
 * but its how WebGL will store matrices in memory.
 */
export class TgdMat3 extends Float32Array {
    constructor()
    constructor(mat: TgdMat3)

        m00: number,

        m20: number = 0,
        m21: number = 0,
        m22: number = 1
    ) {
        if (typeof m00 === "number") {
            super([m00, m01, m02, m10, m11, m12, m20, m21, m22])
        } else {
            super(m00)
        }
    }

    transpose(): TgdMat3 {
        let tmp = this.m10
        this.m10 = this.m01
        this.m01 = tmp
        tmp = this.m20
        this.m20 = this.m02
        this.m02 = tmp
        tmp = this.m21
        this.m21 = this.m12
        this.m12 = tmp
        return this
    }

    fromQuat({ x, y, z, w }: TgdQuat): TgdMat3 {
        const x2 = x + x
        const y2 = y + y
        const z2 = z + z
        const xx = x * x2
        const yx = y * x2
        const yy = y * y2
        const zx = z * x2
        const zy = z * y2
        const zz = z * z2
        const wx = w * x2
        const wy = w * y2
        const wz = w * z2

        this.m00 = 1 - yy - zz
        this.m10 = yx - wz
        this.m20 = zx + wy
 
        this.m01 = yx + wz
        this.m11 = 1 - xx - zz
        this.m21 = zy - wx

        this.m02 = zx - wy
        this.m12 = zy + wx
        this.m22 = 1 - xx - yy

        return this
    }

    toAxis(axisX: TgdVec3, axisY: TgdVec3, axisZ: TgdVec3) {
        this.toAxisX(axisX)
        this.toAxisY(axisY)
        return this.toAxisZ(axisZ)
    }

    toAxisX(axisX: TgdVec3) {
        axisX.x = this.m00
        axisX.y = this.m01
        axisX.z = this.m02
        return this
    }
 
    toAxisY(axisY: TgdVec3) {
        axisY.x = this.m10
        axisY.y = this.m11
        axisY.z = this.m12
        return this
    }

    toAxisZ(axisZ: TgdVec3) {
        axisZ.x = this.m20
        axisZ.y = this.m21
        axisZ.z = this.m22
        return this
    }

    scale(factor: number): this {

        return this
    }
